      onFileChange(e) {
        var cloudname = 'dz7ni7eqi';
        var uploadpreset = 'gwhxdpwq';

        const url = "https://api.cloudinary.com/v1_1/" + cloudname + "/image/upload";

        const file = e.target.files[0];
        this.url = URL.createObjectURL(file);

        var data = new FormData();
        data.append('upload_preset', uploadpreset);
        data.append('file', file);
        data.append('cloud_name', cloudname);
        
        fetch(url, {
          method: "POST",
          body: data
        })
          .then((response) => {
            return response.json();
          })
          .then((data) => {
            var http_string = data.url;
            let https_string = http_string.replace("http:", "https:");
            document.getElementById("url").innerHTML += https_string;
            this.imageurl = https_string;
          });

      }
